var cardHtml = function(cover_image_url, release_day, title , seriesID, currentEpisode){
  return '<div class="card small hoverable"> \
            <div class="card-image"> \
                <img src="' + cover_image_url +'"  width="300" height="250"> \
                <span class="card-title grey-text text-lighten-4">' + release_day + '</span> \
            </div> \
            <div class="card-content"> \
                <p class="truncate">' + title + '</p> \
            </div> \
             <!-- Dropdown Structure --> \
            <div class="card-action"> \
          <button class="waves-effect waves-light btn right ep-done left"  \
      series-id="' + seriesID + '">#<strong id="ep-number-' + seriesID + '"> \
      '+ currentEpisode + '</strong><i class="material-icons left">done</i></button> \
            </div></div>';
};
$.ajax({
  url: '/tracker/get_series_as_json/',
  type: 'GET',
  data: {csrfmiddlewaretoken: document.getElementsByName('csrfmiddlewaretoken')[0].value},
  dataType: 'json',
  success: function(response){
    // clear deck
    $("#series-deck").html('');
    for (var key in response){
      if(response.hasOwnProperty(key)){
        var series = response[key];
        $('#series-deck').append(cardHtml(series.cover_image_url, 
                                          series.release_day, 
                                          series.title , 
                                          series.id, 
                                          series.current_episode));
      }
    }
    
  },
  error: function(response){
    console.log(response);
  }
  
  });
